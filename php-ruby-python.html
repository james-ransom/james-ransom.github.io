Cloud Instance Retrieval Function PHP Function: 
        public function getAllInstances($prefix, $enforcePrefix = true): ?array {
            $cmd = "gcloud compute instances list --filter='status=RUNNING AND name~.*$prefix.*' | awk '{print $1}' | grep -v NAME";
            $group_sets = $this->runCommand($cmd);
            $set = array();
            foreach ($group_sets as $group_set) {
                if ($enforcePrefix && strpos($group_set, 'replica') === false) {
                    continue;
                }
                if (trim($group_set) != '') {
                    $set[] = $group_set;
                }
            }
            return $set;
        }
Ruby Equivalent: 
        def get_all_instances(prefix, enforce_prefix = true)
            cmd = "gcloud compute instances list --filter='status=RUNNING AND name~.*#{prefix}.*' | awk '{print $1}' | grep -v NAME"
            group_sets = `#{cmd}`.split("\n")
            set = []
            group_sets.each do |group_set|
                next if enforce_prefix && !group_set.include?('replica')
                set << group_set unless group_set.strip.empty?
            end
            set
        end
Python Equivalent: 
        import subprocess

        def get_all_instances(prefix, enforce_prefix=True):
            cmd = "gcloud compute instances list --filter='status=RUNNING AND name~.*{}.*' | awk '{{print $1}}' | grep -v NAME".format(prefix)
            group_sets = subprocess.check_output(cmd, shell=True).decode().split('\n')
            set = []
            for group_set in group_sets:
                if enforce_prefix and 'replica' not in group_set:
                    continue
                if group_set.strip():
                    set.append(group_set)
            return set
